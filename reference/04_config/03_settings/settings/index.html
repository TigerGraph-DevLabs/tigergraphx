    Settings

    Bases: BaseConfig

    Application settings, including configurations for vector databases, LLMs, embeddings, and chat models.

    vector_db: LanceDBConfig | BaseVectorDBConfig = Field(description='Configuration for the vector database.') class-attribute instance-attribute

    llm: OpenAIConfig | BaseLLMConfig = Field(description='Configuration for the language model.') class-attribute instance-attribute

    embedding: OpenAIEmbeddingConfig | BaseEmbeddingConfig = Field(description='Configuration for the embedding model.') class-attribute instance-attribute

    chat: OpenAIChatConfig | BaseChatConfig = Field(description='Configuration for the chat model.') class-attribute instance-attribute
